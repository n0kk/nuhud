//Health Bonus Pulse
event HudHealthBonusPulse
{
	Animate	PlayerStatusHealthBonusImage 	Alpha		"255"		Linear 0.0 0.2
	Animate	PlayerStatusHealthBonusImage 	Alpha		"0"			Linear 0.2 0.4
	Animate PlayerStatusHealthBuffBG		Alpha		"255"		Linear 0.0 0.0
	
	Animate PlayerStatusHealthValue 			FgColor 	"Health Buff" 	Linear 0.0 0.0
	Animate PlayerStatusHealthValueSpec 		FgColor 	"Health Buff" 	Linear 0.0 0.0
	Animate PlayerStatusHealthValueSpecgui 		FgColor 	"Health Buff"	Linear 0.0 0.0
	Animate PlayerStatusHealthValueFreeze 		FgColor 	"Health Buff" 	Linear 0.0 0.0
	

	RunEvent HudHealthBonusPulseLoop	0.4
}

// call to loop HudHealthBonusPulse
event HudHealthBonusPulseLoop
{
	RunEvent HudHealthBonusPulse 0.0
}

event HudHealthBonusPulseStop
{
	StopEvent HudHealthBonusPulse 0.0
	StopEvent HudHealthBonusPulseLoop 0.0
	
	Animate PlayerStatusHealthBuffBG			Alpha		"0"			Accel 0.0 0.0
	
	Animate PlayerStatusHealthValue 			FgColor 	"Health" 	Accel 0.0 0.0
	Animate PlayerStatusHealthValueSpec 		FgColor 	"Health" 	Accel 0.0 0.0
	Animate PlayerStatusHealthValueSpecgui 		FgColor		"Health" 	Accel 0.0 0.0
	Animate PlayerStatusHealthValueFreeze 		FgColor 	"Health"	Accel 0.0 0.0
}

// Flash the medic charge hud when we have full charge

event HudMedicCharged
{
	Animate	ChargeLabel 	FgColor		"Ubercharge 1"		Linear 0.0 0.1
	Animate	ChargeLabel 	FgColor		"Ubercharge 2"		Linear 0.3 0.4
	
	Animate	ChargeMeter 	FgColor		"Ubercharge 1"		Linear 0.0 0.1
	Animate	ChargeMeter 	FgColor		"Ubercharge 2"		Linear 0.3 0.4

	RunEvent HudMedicChargedLoop	0.6
}

{

}

{

	Animate	ChargeLabel 	FgColor		"ChargePercent"			Linear 0.0 0.0001
